# Get a function run  {{ tag: 'GET', label: '/v1/runs/{runID}' }}

Returns a single function run by its run ID



<Row>
  <Col>

    ## Request

    <Properties>
      <Property name="runID" type="string" required>
        The unique identifier for the run (a ULID)
      </Property>
    </Properties>

    ## Response

    <Properties>
      <Property name="data" type="object" attribute>
        <Properties nested={true}>
          <Property name="run_id" type="string" attribute>
            The unique identifier for the run (a ULID)
          </Property>
          <Property name="event_id" type="string" attribute>
            The internal unique identifier of the event that triggered the function (a ULID)
          </Property>
          <Property name="status" type="string" attribute>
            Values: `Running`, `Completed`, `Failed`, `Pending`
          </Property>
          <Property name="run_started_at" type="ISO8601 Timestamp" attribute>
            The timestamp that the run started
          </Property>
          <Property name="ended_at" type="ISO8601 Timestamp" attribute>
            The timestamp that the run ended
          </Property>
          <Property name="function_id" type="string" attribute>
            Function uuid
          </Property>
          <Property name="function_version" type="Number" attribute>
            The version of the function that was run
          </Property>
          <Property name="environment_id" type="string" attribute>
            Environment uuid
          </Property>
        </Properties>
      </Property>
    </Properties>

  </Col>
